
DROP DATABASE IF EXISTS CollegeDB;
CREATE DATABASE CollegeDB;
USE CollegeDB;


CREATE TABLE Student (
    StudID INT,
    Name VARCHAR(50),
    Address VARCHAR(100),
    Marks INT
);


CREATE TABLE Employee (
    EmployeeID INT,
    Name VARCHAR(50),
    Address VARCHAR(100),
    Salary DECIMAL(10,2),
    DateOfJoining DATE,
    Department VARCHAR(50)
);


CREATE TABLE Weather (
    CityID INT,
    CityName VARCHAR(50),
    MinTemp FLOAT,
    MaxTemp FLOAT
);
